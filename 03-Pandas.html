Pandas — Data Analysis Powerhouse

Pandas is your most important tool as a data analyst. It provides the DataFrame — a table of data like Excel but programmable. You'll use it every single day to load, clean, transform, and summarise datasets.

1. DataFrame & Series

A Series is a single column of data (like a list with labels). A DataFrame is a collection of Series — essentially a table with named rows and columns.

python
import pandas as pd

# Series — a single column
sales = pd.Series([1200, 3400, 2800, 5100],
                  index=["Jan", "Feb", "Mar", "Apr"])
print(sales["Feb"])    # 3400
print(sales.mean())    # 3125.0

# DataFrame — a table
df = pd.DataFrame({
    "name":       ["Aisha", "Raj", "Priya", "Dev"],
    "department": ["Sales", "IT",    "HR",    "Sales"],
    "salary":     [55000,  72000, 48000,  61000],
    "experience": [3,       5,      2,       4]
})

print(df.shape)         # (4, 4) — 4 rows, 4 columns
print(df.dtypes)        # data type of each column
print(df.head(2))       # first 2 rows
print(df.tail(2))       # last 2 rows
print(df.info())        # summary: types, null counts
print(df.describe())    # stats: mean, std, min, max

2. Reading & Writing Data

python
# Read CSV (most common)
df = pd.read_csv("sales_data.csv")

df = pd.read_csv("data.csv", index_col="id")   # set column as index
df = pd.read_csv("data.csv", nrows=1000)      # read first 1000 rows
df = pd.read_csv("data.csv", skiprows=2)      # skip header rows
df = pd.read_csv("data.csv", na_values=["N/A", "-", ""])

# Read Excel
df = pd.read_excel("report.xlsx", sheet_name="Sales")

# Read JSON
df = pd.read_json("data.json")

# Write to file
df.to_csv("output.csv", index=False)        # no row numbers
df.to_excel("report.xlsx", index=False)
df.to_json("data.json", orient="records")

3. Selecting Data — loc & iloc

loc selects by label (column name, index name). iloc selects by position (integer index, like a list). This distinction trips up every beginner — master it early.

python
# Select a single column → returns Series
df["salary"]
df.salary         # same, but avoid if column name has spaces

# Select multiple columns → returns DataFrame
df[["name", "salary"]]

# loc — label-based [rows, columns]
df.loc[0, "name"]            # "Aisha" — row 0, column "name"
df.loc[0:2, "name":"salary"]  # rows 0-2, cols name to salary
df.loc[:, "salary"]           # all rows, salary column

# loc with boolean condition
df.loc[df["salary"] > 55000]
df.loc[df["department"] == "Sales"]
df.loc[(df["salary"] > 50000) & (df["experience"] >= 3)]

# iloc — integer position [rows, columns]
df.iloc[0]              # first row
df.iloc[0, 2]           # row 0, column index 2
df.iloc[1:3, 0:2]       # rows 1-2, first 2 columns
df.iloc[-1]             # last row
df.iloc[:, -1]          # last column

4. Data Cleaning — Missing Values

Real datasets are always messy. Handling nulls (NaN) is one of the most frequent tasks.

python
# Detect nulls
df.isnull()              # True/False for each cell
df.isnull().sum()        # count nulls per column
df.isnull().sum() / len(df) * 100   # % null per column

# Drop rows/columns with nulls
df.dropna()                           # drop any row with ANY null
df.dropna(subset=["salary"])         # drop only if salary is null
df.dropna(how="all")                 # drop only if ALL columns null
df.dropna(thresh=3)                  # keep rows with ≥ 3 non-nulls

# Fill nulls
df["salary"].fillna(0)               # replace with 0
df["salary"].fillna(df["salary"].mean())  # replace with mean
df["city"].fillna("Unknown")         # replace with string
df["salary"].fillna(method="ffill")  # forward fill
df["salary"].fillna(method="bfill")  # backward fill

# Remove duplicates
df.duplicated()                       # True if row is duplicate
df.drop_duplicates()                  # remove duplicates
df.drop_duplicates(subset=["name"])  # duplicate by specific column

5. GroupBy — Aggregate by Category

GroupBy is the pandas equivalent of SQL's GROUP BY or Excel's PivotTable. It splits data into groups, applies a function, and combines the results.

python
# Average salary by department
df.groupby("department")["salary"].mean()

# Multiple aggregations at once
df.groupby("department").agg(
    avg_salary=("salary", "mean"),
    max_salary=("salary", "max"),
    headcount=("name", "count")
)

# Group by multiple columns
df.groupby(["department", "city"])["salary"].mean()

# transform — keep original shape (add column)
df["dept_avg_salary"] = df.groupby("department")["salary"].transform("mean")

# Salary above department average?
df["above_avg"] = df["salary"] > df["dept_avg_salary"]

# value_counts — count occurrences
df["department"].value_counts()
df["department"].value_counts(normalize=True)  # as proportions

6. Merge & Join

Combine multiple DataFrames — equivalent to SQL JOINs. This is essential when your data lives in multiple tables (employees, departments, sales records).

python
employees = pd.DataFrame({
    "emp_id": [1, 2, 3, 4],
    "name":   ["Aisha", "Raj", "Priya", "Dev"],
    "dept_id": [10, 20, 10, 30]
})

departments = pd.DataFrame({
    "dept_id": [10, 20, 30],
    "dept_name": ["Sales", "IT", "HR"]
})

# Inner join — only matching rows
merged = pd.merge(employees, departments, on="dept_id", how="inner")

# Left join — all rows from left, match from right
merged = pd.merge(employees, departments, on="dept_id", how="left")

# Different column names in each table
pd.merge(df1, df2, left_on="employee_id", right_on="id")

# Concatenate — stack DataFrames
q1 = pd.DataFrame({"month": ["Jan","Feb"], "sales": [1000, 1200]})
q2 = pd.DataFrame({"month": ["Apr","May"], "sales": [1400, 1100]})
full_year = pd.concat([q1, q2], ignore_index=True)

7. apply & map — Custom Transformations

python
# apply — run a function on each value
df["salary_inr"] = df["salary"].apply(lambda x: f"₹{x:,}")

# apply with a custom function
def grade(score):
    if score >= 90: return "A"
    elif score >= 75: return "B"
    elif score >= 60: return "C"
    else: return "F"

df["grade"] = df["score"].apply(grade)

# apply across rows (axis=1)
df["full_label"] = df.apply(
    lambda row: f"{row['name']} ({row['department']})", axis=1
)

# map — replace values using a dict
df["dept_code"] = df["department"].map({
    "Sales": "SL", "IT": "IT", "HR": "HR"
})

# String methods — .str accessor
df["name"].str.upper()
df["name"].str.strip()
df["email"].str.contains("@gmail")
df["name"].str.replace("  ", " ")

8. Pivot Table

python
sales_data = pd.DataFrame({
    "region":  ["North", "South", "North", "South"],
    "product": ["Laptop", "Laptop", "Phone", "Phone"],
    "sales":   [5000, 3000, 2000, 4500]
})

pivot = sales_data.pivot_table(
    values="sales",       # what to aggregate
    index="region",       # rows
    columns="product",    # columns
    aggfunc="sum",         # function: sum, mean, count
    fill_value=0           # replace NaN with 0
)
print(pivot)
# product   Laptop  Phone
# region
# North       5000   2000
# South       3000   4500
