Matplotlib & Seaborn

Data storytelling through visualisation. Matplotlib gives you full control over every pixel. Seaborn adds a beautiful layer on top for fast, statistical charts. Together, they cover every chart you'll ever need as a data analyst.

1. Matplotlib Basics — Figure & Axes

Every Matplotlib chart has two core objects: the Figure (the canvas) and the Axes (the plot area within the canvas). Understanding this separates clean, professional charts from messy ones.

python
import matplotlib.pyplot as plt

# Create figure + axes (recommended approach)
fig, ax = plt.subplots(figsize=(8, 5))   # width=8in, height=5in

months = ["Jan", "Feb", "Mar", "Apr", "May"]
sales  = [12000, 15000, 13500, 17000, 19500]

ax.plot(months, sales, color="#4f8ef7", linewidth=2, marker="o")

# Labels & formatting
ax.set_title("Monthly Sales 2024", fontsize=16, fontweight="bold")
ax.set_xlabel("Month", fontsize=12)
ax.set_ylabel("Sales (₹)", fontsize=12)
ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda x, _: f"₹{x/1000:.0f}K"))
ax.grid(axis="y", alpha=0.3)
ax.spines["top"].set_visible(False)
ax.spines["right"].set_visible(False)

plt.tight_layout()
plt.savefig("sales_chart.png", dpi=150, bbox_inches="tight")
plt.show()

2. Bar Charts

python
departments = ["Sales", "IT", "HR", "Finance"]
avg_salary  = [58000, 74000, 50000, 65000]

fig, ax = plt.subplots(figsize=(8, 5))

bars = ax.bar(departments, avg_salary,
              color=["#4f8ef7", "#00d4aa", "#f7a84f", "#f74f82"],
              edgecolor="white", linewidth=0.8)

# Add value labels on top of bars
for bar, val in zip(bars, avg_salary):
    ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 500,
            f"₹{val/1000:.0f}K", ha="center", va="bottom", fontsize=11)

ax.set_title("Average Salary by Department", fontsize=15, fontweight="bold")
ax.set_ylabel("Average Salary (₹)")

# Horizontal bar chart
ax.barh(departments, avg_salary)

3. Scatter Plots

python
import numpy as np

experience = np.array([1, 2, 3, 4, 5, 6, 7, 8])
salary     = np.array([35, 42, 50, 58, 65, 74, 80, 92]) * 1000

fig, ax = plt.subplots(figsize=(8, 5))
ax.scatter(experience, salary, s=100, color="#4f8ef7", alpha=0.8, edgecolors="white")

# Add trend line
z = np.polyfit(experience, salary, 1)
p = np.poly1d(z)
ax.plot(experience, p(experience), "--", color="#f74f82", linewidth=1.5, label="Trend")

ax.set_title("Experience vs Salary", fontsize=15)
ax.set_xlabel("Years of Experience")
ax.set_ylabel("Salary (₹)")
ax.legend()
plt.tight_layout()
plt.show()

5. Seaborn — Statistical Visualisation

Seaborn wraps Matplotlib with a beautiful default theme and adds chart types tailored for statistical analysis — pair plots, violin plots, box plots, and heatmaps.

python
import seaborn as sns
import matplotlib.pyplot as plt

# Set theme — whitegrid is clean for presentations
sns.set_theme(style="whitegrid", palette="muted")

# Load built-in dataset
tips = sns.load_dataset("tips")

# Box plot — show distribution by category
sns.boxplot(data=tips, x="day", y="total_bill", palette="Set2")
plt.title("Bill Distribution by Day")
plt.show()

# Violin plot — combines boxplot + density
sns.violinplot(data=tips, x="day", y="total_bill", hue="sex")
plt.show()

# Pair plot — all pairwise relationships
iris = sns.load_dataset("iris")
sns.pairplot(iris, hue="species")
plt.show()

# Heatmap — correlations
corr = tips.select_dtypes("number").corr()
sns.heatmap(corr, annot=True, fmt=".2f", cmap="coolwarm")
plt.title("Correlation Matrix")
plt.show()

# Count plot — bar chart for categorical
sns.countplot(data=tips, x="day", order=["Thur","Fri","Sat","Sun"])

# Scatter with regression line
sns.regplot(data=tips, x="total_bill", y="tip")
plt.show()

6. Saving Figures

python
# Save in multiple formats
fig.savefig("chart.png", dpi=300, bbox_inches="tight")   # high-res PNG
fig.savefig("chart.pdf", bbox_inches="tight")             # vector PDF
fig.savefig("chart.svg")                                   # SVG for web

# Tip: always call savefig BEFORE plt.show()
# plt.show() clears the figure buffer
