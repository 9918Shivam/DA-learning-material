1. Variables & Data Types

A variable is a named container that stores a value. Python is dynamically typed — you don't declare the type; Python infers it automatically.

python
# Integers — whole numbers
age = 25
year = 2024

# Floats — decimal numbers
salary = 45000.50
pi = 3.14159

# Strings — text
name = "Priya Sharma"
city = 'Lucknow'

# Booleans — True or False
is_analyst = True
is_manager = False

# NoneType — absence of value
bonus = None

# Check the type of any variable
print(type(salary))   # <class 'float'>
print(type(name))     # <class 'str'>

Type Conversion

Convert between types using built-in functions: int(), float(), str(), bool().

python
revenue_str = "15000"
revenue_int = int(revenue_str)   # "15000" → 15000
revenue_flt = float(revenue_str) # "15000" → 15000.0

score = 0
print(bool(score))  # False — 0 is falsy
print(bool(5))      # True — any non-zero is truthy

2. Collections — Lists, Tuples, Dicts, Sets

Lists — ordered, mutable

python
sales = [1200, 3400, 2800, 5100, 4200]

# Indexing (0-based)
print(sales[0])    # 1200 — first element
print(sales[-1])   # 4200 — last element

# Slicing [start:stop:step]
print(sales[1:4])   # [3400, 2800, 5100]
print(sales[::2])   # [1200, 2800, 4200] — every other

# Modifying
sales.append(6000)       # add to end
sales.insert(0, 900)     # insert at index 0
sales.remove(3400)       # remove first occurrence
popped = sales.pop()      # remove & return last

# Useful methods
print(len(sales))         # number of items
print(sum(sales))         # total
print(max(sales))         # highest value
print(sorted(sales))      # returns new sorted list

Dictionaries — key-value pairs

Dictionaries are the most important data structure for data analysts — JSON responses, config data, and row records all look like dicts.

python
employee = {
    "name": "Rahul",
    "age": 28,
    "department": "Analytics",
    "salary": 72000
}

# Access
print(employee["name"])              # Rahul
print(employee.get("bonus", 0))      # 0 — safe access, no KeyError

# Add / update
employee["city"] = "Lucknow"
employee["salary"] = 78000

# Iterate
for key, value in employee.items():
    print(f"{key}: {value}")

# All keys / values
print(employee.keys())
print(employee.values())

3. Loops — for & while

python
# for loop — iterate over a sequence
months = ["Jan", "Feb", "Mar"]
for month in months:
    print(f"Processing: {month}")

# range() — generate a sequence of numbers
for i in range(1, 6):          # 1, 2, 3, 4, 5
    print(f"Week {i}")

# enumerate — get index + value together
products = ["Laptop", "Phone", "Tablet"]
for idx, product in enumerate(products, start=1):
    print(f"{idx}. {product}")

# while loop — repeat while condition is True
count = 0
while count < 5:
    print(f"Count: {count}")
    count += 1

# break / continue
for num in range(10):
    if num == 3: continue   # skip 3
    if num == 7: break     # stop at 7
    print(num)

4. Functions

Functions let you group reusable logic. In data analysis, you'll write functions to clean columns, calculate KPIs, and transform data.

python
# Basic function
def greet(name):
    return f"Hello, {name}!"

print(greet("Anjali"))   # Hello, Anjali!

# Default arguments
def calculate_tax(income, rate=0.30):
    return income * rate

print(calculate_tax(100000))        # 30000.0 (uses default rate)
print(calculate_tax(100000, 0.20))  # 20000.0

# *args — variable number of positional arguments
def total_sales(*amounts):
    return sum(amounts)

print(total_sales(100, 200, 300))   # 600

# **kwargs — variable keyword arguments
def display_info(**info):
    for k, v in info.items():
        print(f"  {k}: {v}")

display_info(name="Riya", role="Analyst", city="Delhi")

# Lambda — anonymous one-line functions
double = lambda x: x * 2
print(double(7))   # 14

# Often used with map() and filter()
nums = [1, 2, 3, 4, 5]
evens = list(filter(lambda x: x % 2 == 0, nums))   # [2, 4]
doubled = list(map(lambda x: x * 2, nums))           # [2,4,6,8,10]

5. List Comprehensions

A concise, Pythonic way to build lists. You'll use these constantly in data analysis to transform or filter collections in one line.

python
# Syntax: [expression for item in iterable if condition]

# Basic — square every number
squares = [x**2 for x in range(1, 6)]
print(squares)   # [1, 4, 9, 16, 25]

# With condition — only even numbers
evens = [x for x in range(10) if x % 2 == 0]
print(evens)   # [0, 2, 4, 6, 8]

# Real example: extract city names from a list of dicts
employees = [
    {"name": "Neha", "city": "Mumbai"},
    {"name": "Raj",  "city": "Delhi"},
    {"name": "Mia",  "city": "Mumbai"}
]
mumbai_staff = [e["name"] for e in employees if e["city"] == "Mumbai"]
print(mumbai_staff)   # ['Neha', 'Mia']

# Dict comprehension
prices = {"Apple": 100, "Mango": 60, "Banana": 20}
discounted = {k: v * 0.9 for k, v in prices.items()}
print(discounted)   # {'Apple': 90.0, 'Mango': 54.0, 'Banana': 18.0}

6. File I/O — Reading & Writing Files

As a data analyst, you'll constantly read CSVs, write output files, and process text data. Python makes this simple with built-in file handling.

python
# Write to a file
with open("report.txt", "w") as f:
    f.write("Sales Report\n")
    f.write("Total: ₹45,000\n")

# Read entire file
with open("report.txt", "r") as f:
    content = f.read()
    print(content)

# Read line by line (memory-efficient for large files)
with open("report.txt", "r") as f:
    for line in f:
        print(line.strip())

# Append to existing file
with open("report.txt", "a") as f:
    f.write("Updated: 2024-01-15\n")

# Read a CSV manually (before learning pandas)
import csv
with open("sales.csv", "r") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row["product"], row["amount"])
💡
Always use with statements

The with open(...) as f: pattern automatically closes the file even if an error occurs. Never use f = open(...) without a matching f.close().

7. Error Handling — try / except

Real-world data is messy. Files are missing, values are null, formats are wrong. Error handling lets your code respond gracefully instead of crashing.

python
# Basic try/except
try:
    result = 100 / 0
except ZeroDivisionError:
    print("Cannot divide by zero!")

# Multiple exception types
def safe_convert(value):
    try:
        return int(value)
    except ValueError:
        print(f"'{value}' is not a valid number")
        return None
    except TypeError:
        print("Input cannot be None")
        return None

print(safe_convert("abc"))   # 'abc' is not a valid number → None
print(safe_convert("42"))    # 42

# else — runs if no exception occurred
# finally — always runs (cleanup)
try:
    f = open("data.csv")
    data = f.read()
except FileNotFoundError:
    print("File not found — check the path")
else:
    print("File read successfully")
finally:
    print("Attempt complete")   # always runs

8. Modules & Imports

python
# Standard library modules
import math
print(math.sqrt(144))       # 12.0
print(math.ceil(4.2))        # 5

import random
print(random.randint(1, 100))   # random integer 1–100
print(random.choice(["A", "B", "C"]))

from datetime import datetime, date
today = date.today()
print(today)                  # 2024-01-15
print(today.strftime("%d %B %Y"))  # 15 January 2024

# Third-party (install with: pip install numpy pandas)
import numpy as np        # alias 'np' is convention
import pandas as pd       # alias 'pd' is convention

# Import specific function
from os.path import exists, join
if exists("data.csv"):
    print("File found!")
